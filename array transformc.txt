#include <stdio.h>

int main() {
	int t;
	scanf("%d",&t);
	while(t--){
	    int n,k;
	    scanf("%d%d",&n,&k);
	    int a[1001];
	    int i;
	    for(i=0;i<n;i++)
	    scanf("%d",&a[i]);
	    int count=0;
	    int rem=k+1;
	    if(n==2)
	    {
	        printf("YES\n");
	        continue;
	    }
	    for(i=0;i<n;i++){
	        if((a[i+1]-a[i])%rem==0)
	        count++;
	    }
	    if(count>=n-1)
	    printf("YES\n");
	    else
	    printf("NO\n");
	    
	}
	return 0;
}
